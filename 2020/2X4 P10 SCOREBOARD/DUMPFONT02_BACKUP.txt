


;==========	DUMP FONT 0X02===========
DMP_FONT02

	MOVLW		0X01
	MOVWF		FSR1H,0

	MOVLW		0X20
	MOVWF		FSR1L,0

	MOVLW		0
	MOVWF		BTCOUNT	

	MOVLW		.16
	MOVWF		TRAMCOUNT

DMP_F20


	CALL	DUMP_FONT02

	MOVFF	WREG,INDF1

	INCF	FSR1L
	INCF	BTCOUNT

	CALL	DUMP_FONT02

	MOVFF	WREG,INDF1
	INCF	FSR1L
	INCF	FSR1L
	INCF	FSR1L
	INCF	BTCOUNT
	


	DECFSZ	TRAMCOUNT,F
	GOTO	DMP_F20
	
	CALL	ADJ_CURSORX
	CALL	ADJ_CURSORY
	CALL	MOV_TEMPRAM2DRAM
	CALL	CLR_TEMPRAM


RETURN
